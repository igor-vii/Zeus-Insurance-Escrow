Шаг 1: Создать сервис эскроу
Файл: api-server/src/services/escrow.ts

typescript
import { ethers } from "ethers";
import { ZeusSDK } from "@zeus/sdk";

export async function createEscrowAgreement(params: {
  executor: string;
  amount: bigint;
  timeout: number;
}): Promise<{ agreementId: number; txHash: string }> {
  const { executor, amount, timeout } = params;

  const provider = new ethers.JsonRpcProvider(
    process.env.RPC_URL_MAINNET || "https://mainnet.base.org"
  );
  const signer = new ethers.Wallet(process.env.SERVER_PRIVATE_KEY!, provider);

  const sdk = new ZeusSDK();
  await sdk.connect("base-mainnet", signer);

  const result = await sdk.escrow.depositAndCreateAgreement(executor, amount, timeout);

  return {
    agreementId: result.agreementId,
    txHash: result.tx.hash,
  };
}
