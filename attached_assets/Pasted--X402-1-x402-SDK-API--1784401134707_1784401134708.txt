📄 ФАЙЛ 3: api-server/src/app.ts (добавить middleware)
typescript
import express from "express";
import { x402Middleware } from "@x402/sdk/express";
import { x402Config } from "./config/x402";
import insuranceRoutes from "./routes/insurance";
import x402Routes from "./routes/x402";

const app = express();

app.use(express.json());

// Подключаем x402 middleware ко всем эндпоинтам страховки
app.use("/api/insurance", x402Middleware(x402Config));

// Роуты
app.use("/api/insurance", insuranceRoutes);
app.use("/api/x402", x402Routes);

export default app;
📄 ФАЙЛ 4: api-server/src/routes/insurance.ts (обновить эндпоинт)
typescript
import { Router } from "express";
import { x402 } from "@x402/sdk";
import { insuranceService } from "../services/insurance";

const router = Router();

// Создание полиса (поддерживает как x402, так и обычные запросы)
router.post("/policy", async (req, res) => {
  try {
    const { seller, amount, timeout, retries } = req.body;
    
    // Проверяем, был ли платеж через x402
    const payment = await x402.getPayment(req);
    
    if (payment) {
      // Автоматический режим: платёж уже обработан через x402
      const policyId = await insuranceService.createPolicyFromX402({
        seller,
        amount,
        timeout,
        retries,
        payment,
      });
      return res.json({ success: true, policyId });
    }
    
    // Ручной режим: вызываем контракт напрямую (через SDK)
    const policyId = await insuranceService.createPolicyDirect({
      seller,
      amount,
      timeout,
      retries,
      signer: req.signer, // из middleware
    });
    
    res.json({ success: true, policyId });
  } catch (error) {
    console.error("Failed to create policy:", error);
    res.status(500).json({ error: error.message });
  }
});

export default router;
📄 ФАЙЛ 5: api-server/src/routes/x402.ts
typescript
import { Router } from "express";
import { x402 } from "@x402/sdk";

const router = Router();

// Эндпоинт для верификации статуса платежа
router.get("/verify/:paymentId", async (req, res) => {
  try {
    const { paymentId } = req.params;
    const status = await x402.getPaymentStatus(paymentId);
    res.json({ success: true, status });
  } catch (error) {
    res.status(500).json({ error: error.message });
  }
});

// Эндпоинт для получения информации о платеже (для агентов)
router.get("/info", async (req, res) => {
  res.json({
    network: "base-sepolia",
    asset: "0x036CbD53842c5426634e7929541eC2318f3dCF7e",
    payTo: process.env.ZEUS_TREASURY,
    supported: true,
  });
});

export default router;
