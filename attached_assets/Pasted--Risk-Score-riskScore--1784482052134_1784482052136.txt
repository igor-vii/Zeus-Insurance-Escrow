Шаг 4: Добавить обновление Risk Score после выплаты
Файл: api-server/src/services/riskUpdater.ts (новый файл)

typescript
import { db } from "../db/client";
import { policies } from "../db/schema";
import { eq, and } from "drizzle-orm";
import { updateRiskScore } from "./pricing";

/**
 * Update Risk Score for a seller after a payout or completion.
 * Called from the event listener when PolicyPaid or PolicyCompleted event is received.
 */
export async function updateSellerRiskScore(
  sellerAddress: string,
  payoutAmount: bigint,
  wasSuccessful: boolean
): Promise<number> {
  try {
    // 1. Получить текущий средний Risk Score продавца
    const result = await db
      .select({ avgRiskScore: sql<number>`AVG(${policies.riskScore})` })
      .from(policies)
      .where(eq(policies.seller, sellerAddress.toLowerCase()));

    const currentAvg = result[0]?.avgRiskScore || 2.0;

    // 2. Рассчитать payoutFactor
    // 0–5.0: 0 = отличный результат (выплаты не было), 5.0 = провал
    const payoutFactor = wasSuccessful ? 0.5 : 4.0;

    // 3. Обновить Risk Score
    const newRiskScore = await updateRiskScore(
      sellerAddress,
      payoutFactor,
      currentAvg
    );

    // 4. Сохранить новый Risk Score в БД
    await db
      .update(policies)
      .set({ riskScore: newRiskScore.toString() })
      .where(
        and(
          eq(policies.seller, sellerAddress.toLowerCase()),
          eq(policies.status, "completed")
        )
      );

    return newRiskScore;
  } catch (error) {
    console.error("Failed to update seller risk score:", error);
    return 2.0; // fallback
  }
}
