import { ethers } from "ethers";
import { ZeusSDK } from "@zeus/sdk";

export async function createPolicyFromX402(params: {
  seller: string;
  amount: bigint;
  timeout: number;
  retries: number;
  payment: any;
}): Promise<{ policyId: number; txHash: string }> {
  const { seller, amount, timeout, retries, payment } = params;

  if (!payment || !payment.verified) {
    throw new Error("x402 payment required");
  }

  // Серверный кошелёк
  const provider = new ethers.JsonRpcProvider(
    process.env.RPC_URL || "https://mainnet.base.org"
  );
  const signer = new ethers.Wallet(
    process.env.SERVER_PRIVATE_KEY!,
    provider
  );

  const sdk = new ZeusSDK();
  await sdk.connect("base-mainnet", signer);

  const result = await sdk.insurance.createPolicy(
    seller,
    amount,
    timeout,
    retries
  );

  return {
    policyId: result.policyId,
    txHash: result.tx.hash,
  };
}
